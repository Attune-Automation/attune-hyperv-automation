Step 1.2 - BETD Install Python

Execute the following script: 

                
if($env:path -like '*python*') {
    Write-Host 'Python found in PATH, skipping.'
} else {
    
    cd C:\Users\{hypervHostUser.user}\Downloads
    
    # The PrependPath is to automatically update the PATH
    # environment variable.
    
    # There is a bug where the %Path% is not automatically set.
    # https://stackoverflow.com/questions/47310884/adding-into-path-var-while-silent-installation-of-python-possible-bug
    .\python-3.9.9.exe /quiet `
        InstallAllUsers=0 `
        PrependPath=1 `
        Include_test=0
}

Step 5 - BETDSFT Delete Access Token FIle

Execute the following script: 

                
$DIRECTORY = "C:\Users\Public\Downloads"
Set-Location $DIRECTORY

# Add the VM name to the Access Token filename so another
# job for another VM does not delete the access token file 
# required by the current job.
$VM='{targetServer.name}'

$FILE = "$VM-{dropboxRefreshToken.user}AccessToken.txt"

if (Test-Path -Path $FILE -PathType Leaf) {
    Remove-Item "$FILE"
    Write-Host "Deleted $DIRECTORY\$FILE"
} else {
    Write-Host "Can't find $DIRECTORY\$FILE, skipping"
}

Step 6 - BETDSFT Pull Access Token File

Execute the following script: 

                
$CLIENT = New-Object System.Net.WebClient

# Original name of the access token file on Attune
$ACCESS_TOKEN_FILE_REMOTE = "{dropboxRefreshToken.user}AccessToken.txt"

# The remote access token file will be at a directory like:
# /home/attune/kickstart/dropboxAccessTokens/Kean

# Add the VM name to the Access Token filename so another
# job for another VM does not delete the access token file 
# required by the current job.
$VM='{targetServer.name}'
$ACCESS_TOKEN_FILE_LOCAL = "$VM-{dropboxRefreshToken.user}AccessToken.txt"

# Save the file to c:\Users\Public\Downloads\

$URL = "http://{attuneOsBuildServer.ip}:8050/dropboxAccessTokens/{dropboxRefreshToken.user}/$ACCESS_TOKEN_FILE_REMOTE"
$PATH = "c:\Users\Public\Downloads\$ACCESS_TOKEN_FILE_LOCAL"

$CLIENT.DownloadFile($URL, $PATH)
