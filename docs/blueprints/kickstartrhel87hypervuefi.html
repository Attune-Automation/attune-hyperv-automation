Step 3 - KS Hyper-V Cleanup

Step 3.1 - KSHURLVM Remove DVD Drive - Generation 2

Connect via RDP:

mstsc /admin /v:{hypervhost}

Login as user {hypervhostuser} and open a command prompt.

Execute the following script: 

                
$VM='{targetServer.name}'

Write-Host "Deleting VM : $VM"

$i=0
$found=$false

Get-VM | ForEach-Object {
   if ($_.Name -eq $VM) {
       Write-Host "VM $VM found"
       $found=$true
       
       if (Get-VMDVDDrive -VMName $VM) {
            Write-Host "Found DVD drive."
            Remove-VMDvdDrive -VMName $VM -ControllerNumber 0 -ControllerLocation 1
           
            if ($? -eq $true) {
                Write-Host "Removed DVD drive from VM $VM sucessfully"
            } else {
                Write-Host "Failed to remove DVD drive from VM $VM"       
            }
       
       } else {
           Write-Host "No DVD drive found, skipping."
       }
       
       

       return
   }
   $i++
}

if ($found -eq $false) {
    Write-Host "No VM, continuing."
}

Step 3.2 - KS Delete ISO On Hyper-V Host

Execute the following script: 

                
$DIRECTORY='{isoFolder}\kickstart_{targetServer.fqn}'

if (Test-Path $DIRECTORY) {
    cd "$DIRECTORY"
    $ISO = "kickstart_{targetServer.fqn}.iso"
    
    $found = $false
    Get-ChildItem | ForEach-Object {
        $filename=$_.Name
        echo "${i}. ${filename}"
        
        if (${filename} -like $ISO) {
            Write-Host "Found ISO file ${filename}"
            $found=$true
            return
        }
        $i++
    }
    
    if ($found -eq $false) {
        Write-Host "No $ISO found at ${DIRECTORY}, skipping"
    } else {
        Write-Host "Deleting $ISO "
        remove-Item $ISO
        if (Test-Path $ISO) {
            Write-Host "Failed"
        } else {
            Write-Host "Success"   
        }
    }
    
    Write-Host "Deleting folder $DIRECTORY"
    cd ..
    remove-Item $DIRECTORY
    if (Test-Path $DIRECTORY) {
        Write-Host "Failed"
    } else {
        Write-Host "Success"   
    }
    
} else {
    Write-Host "$DIRECTORY does not exist, skipping."   
}
