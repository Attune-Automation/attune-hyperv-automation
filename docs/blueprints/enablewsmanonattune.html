Step 1 - EWOA Install gssntlmssp

Connect via ssh:

ssh {linuxrootuser}@{attuneosbuildserver}

Execute the following script: 

                
dnf -y install gssntlmssp

Step 2 - EWOA Install PSWSMan

Execute the following script: 

                
echo "A" | sudo pwsh -Command "Install-Module -Name PSWSMan"

Step 3 - EWOA Check PSWSMan Installed

Execute the following script: 

                
sudo pwsh -Command "Get-Module -ListAvailable" \
    | grep -i "PSWSMan"

Step 4 - EWOA Run Install-WSMan

Execute the following script: 

                
sudo pwsh -Command "Install-WSMan"

Step 5 - EWOA Add Hyper-V Host to /etc/hosts

Execute the following script: 

                
HOSTS_FILE="/etc/hosts"

ENTRY="{hypervHost.ip}   {hypervHost.hostname}   {hypervHost.hostname}.{hypervHost.domain}"

if grep -iq "{hypervHost.hostname}" "${HOSTS_FILE}"
then
    echo "Replacing hypervHost.hostname} entry in ${HOSTS_FILE}"
    echo "${ENTRY}"
    
    sed -i "s/.*{hypervHost.hostname}.*/${ENTRY}/" "${HOSTS_FILE}"
else
    echo "Appending this entry to ${HOSTS_FILE}"
    echo "${ENTRY}"
    
    echo "${ENTRY}" \
        >> "${HOSTS_FILE}"
fi

Step 6 - KSH Create ISO Folder

Connect via RDP:

mstsc /admin /v:{hypervhost}

Login as user {hypervhostuser} and open a command prompt.

Execute the following script: 

                
if (Test-Path "{isoFolder}") {
    Write-Host "ISO folder {isoFolder} exists, good"
} else {
    Write-Host "ISO folder {isoFolder} deos not exist, creating"
    New-Item "{isoFolder}" -ItemType Directory
    if (Test-Path "{isoFolder}") {
        Write-Host "ISO folder {isoFolder} created."
    } else {
        Write-Error "Failed to create ISO folder {isoFolder}" -ErrorAction Stop
    }
}

Step 7 - EWOA Test Copy-Item Small File

Step 7.1 - ETCSF Attune Create Small Test File

Connect via ssh:

ssh {linuxattuneuser}@{attuneosbuildserver}

Execute the following script: 

                
echo "hi" > "/tmp/{smallTestFileName}"

Step 7.2 - ETCSF Test Copy-Item for Small File

Execute the following script: 

                
pwsh <<'EOF'
$SOURCE_FILE_PATH="/tmp/{smallTestFileName}"

# $userName looks like 'ko1hprv1vs2\kooi'
$userName = '{hypervHost.hostname}\{hypervHostUser.user}'

Write-Host "userName = $userName"

$userPassword = '{hypervHostUser.password}'
$secStringPassword = ConvertTo-SecureString $userPassword -AsPlainText -Force

$credObject = New-Object System.Management.Automation.PSCredential `
    ($userName, $secStringPassword)

$session = New-PSSession -ComputerName '{hypervHost.hostname}' `
    -Credential $credObject -Authentication Negotiate

$ProgressPreference = "SilentlyContinue"

Write-Host "Started Copy-Item"

#$ISO_Folder = "C:\\Users\\Public\\Documents\\Hyper-V\\isos"

Copy-Item -Path "$SOURCE_FILE_PATH" -Destination "{isoFolder}\{smallTestFileName}" `
    -ToSession $session

Write-Host "done"
EOF

Step 7.4 - ETCSF Hyper-V Host Remove Small Test File

Connect via RDP:

mstsc /admin /v:{hypervhost}

Login as user {hypervhostuser} and open a command prompt.

Execute the following script: 

                
$DIRECTORY='{isoFolder}'

if (Test-Path $DIRECTORY) {
    cd "$DIRECTORY"

    $found = $false
    Get-ChildItem | ForEach-Object {
        $filename=$_.Name
        echo "${i}. ${filename}"
        
        if (${filename} -like "{smallTestFileName}") {
            Write-Host "Found small file ${filename}"
            $found=$true
            return
        }
        $i++
    }
    
    if ($found -eq $false) {
        Write-Host "No small file {smallTestFileName} found at ${DIRECTORY}, skipping"
    } else {
        Write-Host "Deleting {smallTestFileName} "
        remove-Item "{smallTestFileName}"
        if (Test-Path "{smallTestFileName}") {
            Write-Host "Failed"
        } else {
            Write-Host "Success"   
        }
    }

    
} else {
    Write-Host "$DIRECTORY does not exist, skipping."   
}
